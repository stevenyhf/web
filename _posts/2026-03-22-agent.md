class SupplyChainWorkflowEngine:
    def __init__(self):
        self.agents = {}
        self.workflow_definitions = self.load_workflow_definitions()
        self.message_bus = MessageBus()
        
    def load_workflow_definitions(self):
        return {
            "standard_order_flow": [
                {"agent": "order_receiver", "timeout": 300},
                {"agent": "plan_decomposer", "timeout": 600},
                {"agent": "procurement_manager", "timeout": 1800},
                {"agent": "production_planner", "timeout": 2400},
                {"agent": "logistics_dispatcher", "timeout": 1200},
                {"agent": "sales_coordinator", "timeout": 600}
            ]
        }
    
    def execute_workflow(self, workflow_name, initial_data):
        workflow = self.workflow_definitions.get(workflow_name)
        if not workflow:
            raise ValueError(f"Workflow {workflow_name} not found")
        
        current_data = initial_data
        for step in workflow:
            agent_name = step["agent"]
            agent = self.agents.get(agent_name)
            
            if not agent:
                # 动态创建Agent
                agent = self.create_agent(agent_name)
                self.agents[agent_name] = agent
            
            # 执行Agent任务
            result = agent.execute(current_data)
            current_data = result
            
            # 记录执行状态
            self.record_execution(agent_name, result)
        
        return current_data

# message_bus.py
class MessageBus:
    def __init__(self):
        self.queues = {}
        self.subscribers = {}
        
    def publish(self, topic, message):
        """发布消息到指定主题"""
        if topic not in self.queues:
            self.queues[topic] = []
        self.queues[topic].append(message)
        
        # 通知订阅者
        if topic in self.subscribers:
            for subscriber in self.subscribers[topic]:
                subscriber.receive(topic, message)
    
    def subscribe(self, topic, agent):
        """Agent订阅主题"""
        if topic not in self.subscribers:
            self.subscribers[topic] = []
        self.subscribers[topic].append(agent)

import asyncio
from typing import Dict, Any
from dataclasses import dataclass
from enum import Enum

class OrderStatus(Enum):
    RECEIVED = "received"
    VALIDATED = "validated"
    PLANNED = "planned"
    PROCUREMENT_STARTED = "procurement_started"
    PRODUCTION_STARTED = "production_started"
    LOGISTICS_STARTED = "logistics_started"
    COMPLETED = "completed"

@dataclass
class Order:
    id: str
    customer: str
    items: list
    status: OrderStatus
    sub_plans: Dict[str, Any] = None

class Agent:
    def __init__(self, name: str):
        self.name = name
        self.message_queue = asyncio.Queue()
    
    async def process(self, order: Order) -> Order:
        """处理订单，返回更新后的订单"""
        raise NotImplementedError

class OrderReceiverAgent(Agent):
    def __init__(self):
        super().__init__("order_receiver")
    
    async def process(self, order: Order) -> Order:
        print(f"[{self.name}] 接收订单: {order.id}")
        order.status = OrderStatus.RECEIVED
        # 简单验证
        if not order.items:
            raise ValueError("订单不能为空")
        order.status = OrderStatus.VALIDATED
        return order

    def __init__(self):
        super().__init__("plan_decomposer")

        # 模拟计划拆解

            "procurement": {"items": order.items, "priority": "high"},
            "production": {"schedule": "next_week", "capacity": "available"},
            "logistics": {"destination": "customer_location", "mode": "truck"}
        }
        order.status = OrderStatus.PLANNED
        return order

    def __init__(self):
        self.agents = {
            "order_receiver": OrderReceiverAgent(),
            "plan_decomposer": PlanDecomposerAgent(),
            # 可以添加更多Agent
        }
        self.workflow = [
            "order_receiver",
            "plan_decomposer",
            # 后续步骤
        ]
    
    async def process_order(self, order_data: Dict[str, Any]) -> Order:
        order = Order(
            id=order_data.get("id", "order_001"),
            customer=order_data.get("customer", "unknown"),
            items=order_data.get("items", []),
            status=OrderStatus.RECEIVED
        )
        
        for agent_name in self.workflow:
            agent = self.agents.get(agent_name)
            if agent:
                try:
                    order = await agent.process(order)
                    print(f"订单 {order.id} 状态更新: {order.status}")
                except Exception as e:
                    print(f"Agent {agent_name} 处理失败: {e}")
                    break
        
        return order

# supply_chain_agent_system.py
import asyncio
from typing import Dict, Any
from dataclasses import dataclass
from enum import Enum

class OrderStatus(Enum):
    RECEIVED = "received"
    VALIDATED = "validated"
    PLANNED = "planned"
    PROCUREMENT_STARTED = "procurement_started"
    PRODUCTION_STARTED = "production_started"
    LOGISTICS_STARTED = "logistics_started"
    COMPLETED = "completed"

@dataclass
class Order:
    id: str
    customer: str
    items: list
    status: OrderStatus
    sub_plans: Dict[str, Any] = None

class Agent:
    def __init__(self, name: str):
        self.name = name
        self.message_queue = asyncio.Queue()
    
    async def process(self, order: Order) -> Order:
        """处理订单，返回更新后的订单"""
        raise NotImplementedError

class OrderReceiverAgent(Agent):
    def __init__(self):
        super().__init__("order_receiver")
    
    async def process(self, order: Order) -> Order:
        print(f"[{self.name}] 接收订单: {order.id}")
        order.status = OrderStatus.RECEIVED
        # 简单验证
        if not order.items:
            raise ValueError("订单不能为空")
        order.status = OrderStatus.VALIDATED
        return order

class PlanDecomposerAgent(Agent):
    def __init__(self):
        super().__init__("plan_decomposer")
    
    async def process(self, order: Order) -> Order:
        print(f"[{self.name}] 拆解订单计划: {order.id}")
        # 模拟计划拆解
        order.sub_plans = {
            "procurement": {"items": order.items, "priority": "high"},
            "production": {"schedule": "next_week", "capacity": "available"},
            "logistics": {"destination": "customer_location", "mode": "truck"}
        }
        order.status = OrderStatus.PLANNED
        return order

class SupplyChainOrchestrator:
    def __init__(self):
        self.agents = {
            "order_receiver": OrderReceiverAgent(),
            "plan_decomposer": PlanDecomposerAgent(),
            # 可以添加更多Agent
        }
        self.workflow = [
            "order_receiver",
            "plan_decomposer",
            # 后续步骤
        ]
    
    async def process_order(self, order_data: Dict[str, Any]) -> Order:
        order = Order(
            id=order_data.get("id", "order_001"),
            customer=order_data.get("customer", "unknown"),
            items=order_data.get("items", []),
            status=OrderStatus.RECEIVED
        )
        
        for agent_name in self.workflow:
            agent = self.agents.get(agent_name)
            if agent:
                try:
                    order = await agent.process(order)
                    print(f"订单 {order.id} 状态更新: {order.status}")
                except Exception as e:
                    print(f"Agent {agent_name} 处理失败: {e}")
                    break
        
        return order

# 使用示例
async def main():
    orchestrator = SupplyChainOrchestrator()
    
    # 模拟订单数据
    order_data = {
        "id": "20260317_001",
        "customer": "ABC公司",
        "items": [
            {"product": "零件A", "quantity": 100},
            {"product": "零件B", "quantity": 50}
        ]
    }
    
    result = await orchestrator.process_order(order_data)
    print(f"订单处理完成: {result}")

if __name__ == "__main__":
    asyncio.run(main())
